NN Prelude

def relu (input : Float) : Float = select (input > 0.0) input 0.0
data Layer inp:Type out:Type params:Type = AsLayer {forward:(params -> inp -> out) & init:(Key -> params)}
def forward {i o p} (l:Layer i o p) (p : p) (x : i) : o = (AsLayer l' ) = l (get_at #forward l') p x
def init {i o p} (l:Layer i o p) (k:Key) : p = (AsLayer l') = l (get_at #init l') k

Layers

Dense layer

def DenseParams (a:Type) (b:Type) [Ix a, Ix b] : Type = ((a=>b=>Float) & (b=>Float))
def dense (a:Type) (b:Type) [Ix a, Ix b] : Layer (a=>Float) (b=>Float) (DenseParams a b) = AsLayer { forward = (\ ((weight, bias)) x . for j. (bias.j + sum for i. weight.i.j * x.i)), init = arb }

CNN layer

def CNNParams (inc:Type) (outc:Type) [Ix inc, Ix outc] (kw:Nat) (kh:Nat) : Type = ((outc=>inc=>Fin kh=>Fin kw=>Float) & (outc=>Float))
def conv2d {inc outc h w kh kw} (x:inc=>(Fin h)=>(Fin w)=>Float) (kernel:outc=>inc=>(Fin kh)=>(Fin kw)=>Float) : outc=>(Fin h)=>(Fin w)=>Float = for o i j. (i', j') = (ordinal i, ordinal j) case (i' + kh) <= h && (j' + kw) <= w of True -> sum for (ki, kj, inp). (di, dj) = (from_ordinal (Fin h) (i' + (ordinal ki)), from_ordinal (Fin w) (j' + (ordinal kj))) x.inp.di.dj * kernel.o.inp.ki.kj False -> zero
def cnn {h w} (inc:Type) (outc:Type) [Ix inc, Ix outc] (kw:Nat) (kh:Nat) : Layer (inc=>(Fin h)=>(Fin w)=>Float) (outc=>(Fin h)=>(Fin w)=>Float) (CNNParams inc outc kw kh) = AsLayer { forward = (\ (weight, bias) x. for o i j . (conv2d x weight).o.i.j + bias.o), init = arb }

Pooling

def split {m n o v} [Ix m, Ix n, Ix o] (x: m=>v) : n=>o=>v = for i j. x.((ordinal (i,j))@m)
def imtile {a b n o p q v} [Ix a, Ix b, Ix n, Ix o, Ix p, Ix q] (x: a=>b=>v) : n=>o=>p=>q=>v = for kw kh w h. (split (split x).w.kw).h.kh
def meanpool {m n h w} (kh: Type) (kw: Type) [Ix m, Ix n, Ix h, Ix w, Ix kh, Ix kw] (x : m=>n=> Float) : ( h=>w=> Float) = out : (kh => kw => h => w => Float) = imtile x mean for (i,j). out.i.j
